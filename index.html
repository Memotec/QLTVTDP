function doGet() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < rows.length; i++) { if(rows[i][1]) { result.push({ name: rows[i][1], code: rows[i][2], sn: rows[i][3], loc: rows[i][4] }); } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); // KIỂM TRA NẾU LÀ LỆNH XÓA MỘT HÀNG if (data.action === "DELETE_ROW") { var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { // Tìm hàng có Mã CCDC khớp với mã gửi sang để xóa if (rows[i][2] == data.code) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput("Đã xóa hàng").setMimeType(ContentService.MimeType.TEXT); } // NẾU LÀ LỆNH LƯU/THÊM MỚI (Vẫn giữ ghi đè để đồng bộ toàn bảng khi cần) sheet.clearContents(); sheet.appendRow(["STT", "Thiết bị", "Mã CCDC", "S/N", "Vị trí"]); data.forEach(function(item, index) { sheet.appendRow([index + 1, item.name, item.code, item.sn, item.loc]); }); return ContentService.createTextOutput("Thành công").setMimeType(ContentService.MimeType.TEXT); }
